class Main {
    public static void main(String[] args) {
        int n;
       int sp=0;
       int st=n-1;
       for(int i=0;i<n;i++){
           for(int j=0;j<sp;j++){
               System.out.println(" /t");
               sp--;
           }
           for(int j=0;j<st;j++){
                System.out.println("*/t");
           }
              st++;
           }
       }
    }
